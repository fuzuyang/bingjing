#配置前端路径和后端路径，然后在nginx上写好配置
/home/www/risk/frontend
/home/www/risk

# 备份原配置（以防万一）
sudo cp /etc/nginx/sites-available/zlyf.club /etc/nginx/sites-available/zlyf.club.bak.$(date +%Y%m%d_%H%M%S)

# 删除原配置
sudo rm /etc/nginx/sites-available/zlyf.club

# 创建新的配置文件
sudo tee /etc/nginx/sites-available/zlyf.club << 'EOF'
# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name zlyf.club www.zlyf.club;
    return 301 https://$server_name$request_uri;
}

# HTTPS 配置
server {
    listen 443 ssl http2;
    server_name zlyf.club www.zlyf.club;
    
    # SSL证书配置
    ssl_certificate /etc/ssl/certs/fullchain.pem;
    ssl_certificate_key /etc/ssl/private/privkey.key;
    
    # 修复SSL配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers 'ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384';
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
  
    # ========== FADS子应用 ==========
    location ^~ /fads/ {
        root /home/www/html;
        index index.html;
        try_files $uri $uri/ /fads/index.html;    
    }

    # ========== 主应用 ==========
    location / {
        root /home/www/html;
        index index.html;
        try_files $uri $uri/ /index.html;
    }
    
    # ========== 原有API代理 ==========
    location /api/ {
        proxy_pass http://127.0.0.1:5040;
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        proxy_read_timeout 300s;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_buffering off;
        proxy_request_buffering off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # CORS配置
        add_header Access-Control-Allow-Origin $http_origin always;
        add_header Access-Control-Allow-Credentials true always;
        add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With" always;
        
        # 处理预检请求
        if ($request_method = "OPTIONS") {
            add_header Access-Control-Allow-Origin $http_origin;
            add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS";
            add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With";
            add_header Access-Control-Max-Age 1728000;
            return 204;
        }
    }
    
    # ========== 新增：风险评估项目 ==========
    # 前端静态文件服务
    location /risk/ {
        alias /home/www/risk/frontend/;
        index index.html;
        try_files $uri $uri/ /risk/index.html;
        
        # 缓存静态资源
        location ~* /risk/.*\.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
            expires 30d;
            add_header Cache-Control "public, no-transform";
        }
    }
    
    # 风险评估API代理
    location /risk-api/ {
        rewrite ^/risk-api/(.*) /$1 break;
        proxy_pass http://127.0.0.1:5045;
        
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        proxy_read_timeout 300s;
        
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_buffering off;
        proxy_request_buffering off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # CORS配置
        add_header Access-Control-Allow-Origin $http_origin always;
        add_header Access-Control-Allow-Credentials true always;
        add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With" always;
        
        # 处理预检请求
        if ($request_method = "OPTIONS") {
            add_header Access-Control-Allow-Origin $http_origin;
            add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS";
            add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With";
            add_header Access-Control-Max-Age 1728000;
            return 204;
        }
    }
    
    # ========== 错误页面 ==========
    error_page 404 /404.html;
    location = /404.html {
        root /home/www/html;
        internal;
    }
    
    # ========== 安全头部 ==========
    add_header X-Frame-Options SAMEORIGIN;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    
    # 防止访问隐藏文件
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }
}
EOF

# 检查配置
sudo nginx -t

# 如果配置正确，重载Nginx
if [ $? -eq 0 ]; then
    sudo systemctl reload nginx
    echo "✅ Nginx配置更新成功！"
else
    echo "❌ Nginx配置有错误，请检查"
fi
